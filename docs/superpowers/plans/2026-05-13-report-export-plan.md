```typescript
import pdfMake from 'pdfmake/build/pdfmake';
import pdfFonts from 'pdfmake/build/vfs_fonts';

pdfMake.vfs = pdfFonts.pdfMake.vfs;

export function exportPdfMakeReport(results: any[]) {
  const vulns = results.filter(r => r.type === 'vuln' || r.type === 'tls');
  const docDefinition = {
    content: [
      { text: 'NetProwl Security Report', style: 'header' },
      { text: `Generated: ${new Date().toLocaleString()}`, style: 'subheader' },
      { text: `Total findings: ${results.length}`, style: 'subheader' },
      { ul: vulns.map((v: any) => `[${v.severity?.toUpperCase()}] ${v.template || v.name} — ${v.host || v.ip}:${v.port}`) },
    ],
    styles: {
      header: { fontSize: 18, bold: true, margin: [0, 0, 0, 10] },
      subheader: { fontSize: 12, margin: [0, 5, 0, 5] },
    },
  };
  pdfMake.createPdf(docDefinition).download('netprowl-report.pdf');
}
```

## Task 3: JSON + HTML 导出

**Files:**
- Modify: `netprowl-pc/src/components/ExportPanel.tsx`

- [ ] **Step 1: JSON export**

```tsx
const jsonExport = () => {
  const blob = new Blob([JSON.stringify(results, null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `netprowl-report-${Date.now()}.json`;
  a.click();
  URL.revokeObjectURL(url);
};
```

- [ ] **Step 2: HTML export**

```tsx
const htmlExport = () => {
  const html = `<html><head><title>NetProwl Report</title></head><body>
```
